⚡ Quick Reference - Bun YAML API

1. Load YAML → JS Object

import { YAML } from "bun";

const cfg = YAML.parse(await Bun.file("config/app.yml").text());
// cfg is now a plain JS object

2. Save JS Object → YAML

import { YAML } from "bun";

const yamlStr = YAML.stringify(cfg);
await Bun.write("config/app.yml", yamlStr);

3. Direct Import (Build-time)

// Import entire YAML file
import config from "./config.yaml";

// Named imports
import { database, features } from "./config.yaml";

console.log(config.server.port); // 3000

4. Hot-Reload (Zero Restart)

// Run with: bun --hot server.ts
import config from "./config.yaml";

// Changes to config.yaml automatically reload!

5. Environment Variables

# config.yaml
server:
  port: ${PORT:-3000}
  host: ${HOST:-localhost}

# Automatically interpolated by Bun!

6. Schema Validation with Zod

import { YAML } from "bun";
import { z } from "zod";

const schema = z.object({ 
  port: z.number(),
  host: z.string()
});

const cfg = schema.parse(
  YAML.parse(await Bun.file("config.yml").text())
);
